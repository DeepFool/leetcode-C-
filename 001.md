class Solution{
public:
    vector<int> twoSum(vector<int>& nums, int target)
    {
        vector<int> result;
        map<int,int> hs;

        for(int i=0;i<nums.size();i++)
        {
            if(hs.find(target-nums[i]) == hs.end())
            {
                hs[nums[i]] = i;
            }
            else {
                result.push_back(i);
                result.push_back(hs[target - nums[i]]);
                return result;
            }
        }
        return result;
    }
};
